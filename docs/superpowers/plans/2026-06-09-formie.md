# Formie Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Formie, a Chrome MV3 extension that converts natural language prompts into Google Forms via an animated pink cloud-cat mascot (Blanc), with streaming generation, template matching, LLM fallback cascade, and Slack integration.

**Architecture:** Pure client-side Chrome MV3 extension. Vite + CRXJS for build. Google Forms REST API via `chrome.identity` OAuth brokered through the service worker. LLM cascade (Gemini Flash → Grok → Ollama) uses async generators for streaming. All state in `chrome.storage.local`. Content scripts message the service worker for privileged operations (identity, tabs).

**Tech Stack:** Chrome MV3, Vite 5, @crxjs/vite-plugin ^2.0.0-beta.23, Vitest 1, jsdom, Google Forms REST API v1, Gemini 1.5 Flash, Grok (xAI, OpenAI-compatible), Ollama local

// memory/store.js
const DEFAULTS = {
  keys: { gemini: '', grok: '', slackWebhook: '' },
  memory: {
    fieldSchemas: [],
    history: [],
    savedTemplates: []
  }
};

function get(defaults) {
  return new Promise(resolve => chrome.storage.local.get(defaults, resolve));
}

function set(updates) {
  return new Promise(resolve => chrome.storage.local.set(updates, resolve));
}

export async function getKeys() {
  const data = await get({ keys: DEFAULTS.keys });
  return data.keys;
}

export async function saveKey(provider, value) {
  const data = await get({ keys: DEFAULTS.keys });
  data.keys[provider] = value;
  await set({ keys: data.keys });
}

export async function getMemory() {
  const data = await get({ memory: DEFAULTS.memory });
  return data.memory;
}

export async function saveFormToHistory(entry) {
  const data = await get({ memory: DEFAULTS.memory });
  data.memory.history = [entry, ...data.memory.history].slice(0, 20);
  await set({ memory: data.memory });
}

export async function saveTemplate(name, schema) {
  const data = await get({ memory: DEFAULTS.memory });
  data.memory.savedTemplates.push({ name, schema, createdAt: new Date().toISOString() });
  await set({ memory: data.memory });
}

export async function saveFieldSchema(name, fields) {
  const data = await get({ memory: DEFAULTS.memory });
  const existing = data.memory.fieldSchemas.find(s => s.name === name);
  if (existing) {
    existing.usageCount += 1;
    existing.lastUsed = new Date().toISOString();
  } else {
    data.memory.fieldSchemas.push({ name, fields, usageCount: 1, lastUsed: new Date().toISOString() });
  }
  await set({ memory: data.memory });
}
```

// llm/parser.js
export function tryParsePartialQuestions(buffer) {
  const questions = [];
  const match = buffer.match(/"questions"\s*:\s*\[/);
  if (!match) return questions;

  const afterBracket = buffer.slice(match.index + match[0].length);
  let depth = 0;
  let objStart = -1;
  let inString = false;
  let escapeNext = false;

  for (let i = 0; i < afterBracket.length; i++) {
    const ch = afterBracket[i];
    if (escapeNext) { escapeNext = false; continue; }
    if (ch === '\\' && inString) { escapeNext = true; continue; }
    if (ch === '"') { inString = !inString; continue; }
    if (inString) continue;

    if (ch === '{') {
      if (depth === 0) objStart = i;
      depth++;
    } else if (ch === '}') {
      depth--;
      if (depth === 0 && objStart !== -1) {
        try {
          questions.push(JSON.parse(afterBracket.slice(objStart, i + 1)));
        } catch {}
        objStart = -1;
      }
    }
  }

  return questions;
}

export function parseFormJSON(raw) {
  let cleaned = raw.trim().replace(/^```(?:json)?\n?/, '').replace(/\n?```$/, '');
  const schema = JSON.parse(cleaned);
  if (!schema.title || !Array.isArray(schema.questions)) {
    throw new Error('Invalid form schema: missing title or questions array');
  }
  return schema;
}
```

// llm/providers/gemini.js
const BASE = 'https://generativelanguage.googleapis.com/v1beta';
const MODEL = 'gemini-1.5-flash-latest';

function tagged(message, type) {
  return Object.assign(new Error(message), { type });
}

export const gemini = {
  name: 'gemini',

  async isAvailable(apiKey) {
    return Boolean(apiKey);
  },

  async generate(apiKey, system, user) {
    const res = await fetch(`${BASE}/models/${MODEL}:generateContent?key=${apiKey}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        system_instruction: { parts: [{ text: system }] },
        contents: [{ role: 'user', parts: [{ text: user }] }],
        generationConfig: { responseMimeType: 'application/json' }
      })
    });

    if (res.status === 401 || res.status === 403) throw tagged('Auth failed', 'AUTH_FAILED');
    if (res.status === 429) throw tagged('Rate limited', 'RATE_LIMITED');
    if (!res.ok) throw tagged(`HTTP ${res.status}`, 'NETWORK_ERROR');

    const data = await res.json();
    return data.candidates[0].content.parts[0].text;
  },

  async *streamGenerate(apiKey, system, user) {
    const res = await fetch(`${BASE}/models/${MODEL}:streamGenerateContent?key=${apiKey}&alt=sse`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        system_instruction: { parts: [{ text: system }] },
        contents: [{ role: 'user', parts: [{ text: user }] }]
      })
    });

    if (res.status === 401 || res.status === 403) throw tagged('Auth failed', 'AUTH_FAILED');
    if (res.status === 429) throw tagged('Rate limited', 'RATE_LIMITED');
    if (!res.ok) throw tagged(`HTTP ${res.status}`, 'NETWORK_ERROR');

    const reader = res.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      for (const line of decoder.decode(value).split('\n')) {
        if (!line.startsWith('data: ')) continue;
        const jsonStr = line.slice(6).trim();
        if (jsonStr === '[DONE]') return;
        try {
          const chunk = JSON.parse(jsonStr)?.candidates?.[0]?.content?.parts?.[0]?.text;
          if (chunk) yield chunk;
        } catch {}
      }
    }
  }
};
```

const BASE = 'https://api.x.ai/v1';
const MODEL = 'grok-beta';

function tagged(message, type) {
  return Object.assign(new Error(message), { type });
}

export const grok = {
  name: 'grok',

  async isAvailable(apiKey) {
    return Boolean(apiKey);
  },

  async generate(apiKey, system, user) {
    const res = await fetch(`${BASE}/chat/completions`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${apiKey}` },
      body: JSON.stringify({
        model: MODEL,
        messages: [{ role: 'system', content: system }, { role: 'user', content: user }],
        stream: false
      })
    });

    if (res.status === 401) throw tagged('Auth failed', 'AUTH_FAILED');
    if (res.status === 429) throw tagged('Rate limited', 'RATE_LIMITED');
    if (!res.ok) throw tagged(`HTTP ${res.status}`, 'NETWORK_ERROR');

    const data = await res.json();
    return data.choices[0].message.content;
  },

  async *streamGenerate(apiKey, system, user) {
    const res = await fetch(`${BASE}/chat/completions`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${apiKey}` },
      body: JSON.stringify({
        model: MODEL,
        messages: [{ role: 'system', content: system }, { role: 'user', content: user }],
        stream: true
      })
    });

    if (res.status === 401) throw tagged('Auth failed', 'AUTH_FAILED');
    if (res.status === 429) throw tagged('Rate limited', 'RATE_LIMITED');
    if (!res.ok) throw tagged(`HTTP ${res.status}`, 'NETWORK_ERROR');

    const reader = res.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      for (const line of decoder.decode(value).split('\n')) {
        if (!line.startsWith('data: ')) continue;
        const jsonStr = line.slice(6).trim();
        if (jsonStr === '[DONE]') return;
        try {
          const chunk = JSON.parse(jsonStr)?.choices?.[0]?.delta?.content;
          if (chunk) yield chunk;
        } catch {}
      }

// llm/cascade.js
import { gemini } from './providers/gemini.js';
import { grok } from './providers/grok.js';
import { ollama } from './providers/ollama.js';
import { getKeys } from '../memory/store.js';

async function buildProviderChain() {
  const keys = await getKeys();
  const chain = [];

  if (keys.gemini) chain.push({ provider: gemini, apiKey: keys.gemini });
  if (keys.grok) chain.push({ provider: grok, apiKey: keys.grok });

  if (await ollama.isAvailable()) chain.push({ provider: ollama, apiKey: '' });

  return chain;
}

export async function generateWithCascade(system, user, onProviderError) {
  const chain = await buildProviderChain();

  for (const { provider, apiKey } of chain) {
    try {
      return await provider.generate(apiKey, system, user);
    } catch (err) {
      if (err.type === 'AUTH_FAILED') onProviderError?.(provider.name, 'auth');
      // RATE_LIMITED, NETWORK_ERROR, PARSE_FAILED: silent, try next
    }
  }

  throw Object.assign(new Error('All providers failed'), { type: 'ALL_FAILED' });
}

export async function* streamWithCascade(system, user, onProviderError) {
  const chain = await buildProviderChain();

  for (const { provider, apiKey } of chain) {
    try {
      yield* provider.streamGenerate(apiKey, system, user);
      return;
    } catch (err) {
      if (err.type === 'AUTH_FAILED') onProviderError?.(provider.name, 'auth');
    }
  }

  throw Object.assign(new Error('All providers failed'), { type: 'ALL_FAILED' });
}

  "description": "string",
  "questions": [
    {
      "title": "string",

      "required": true | false,
      "options": ["string"],
      "scale": { "low": 1, "high": 5, "lowLabel": "", "highLabel": "" }
    }
  ]

Rules:
- 4–8 questions unless the user specifies a number
- Pick the most appropriate field type for each question
- required: true for identity/contact fields, false for optional feedback
- Never return anything except the JSON object
- If a template is provided, use it as structural scaffolding but adapt field names and description to the user's prompt`;

export function buildSystemPrompt(memoryContext = null, templateContext = null) {
  let prompt = FORM_BUILDER_SYSTEM_PROMPT;
  if (memoryContext) prompt += `\n\n${memoryContext}`;
  if (templateContext) prompt += `\n\n${templateContext}`;
  return prompt;
}

export function buildMemoryContext(fieldSchemas, history) {
  if (!fieldSchemas.length && !history.length) return null;

  const frequentFields = fieldSchemas
    .sort((a, b) => b.usageCount - a.usageCount)
    .slice(0, 3)
    .flatMap(s => s.fields);

  const uniqueFields = [...new Set(frequentFields)].slice(0, 6);
  const recentForms = history.slice(0, 3).map(h => `"${h.title}"`);

  let ctx = "User's frequent field patterns:";
  if (uniqueFields.length) ctx += ` [${uniqueFields.join(', ')}]`;
  if (recentForms.length) ctx += `\nRecent forms: [${recentForms.join(', ')}]`;
  ctx += '\nAdapt field naming and structure to match their style.';
  return ctx;
}

export function buildTemplateContext(template) {
  return `Template match: "${template.name}". Use this as structural scaffolding:\n${JSON.stringify(template.schema, null, 2)}`;
}

const PATTERNS = {

  quiz: []

function detectType(title) {
  const lower = title.toLowerCase();
  return Object.keys(TYPE_REQUIREMENTS).find(t => lower.includes(t)) ?? 'default';
}

function questionCovers(question, category) {
  const lower = question.title.toLowerCase();
  return PATTERNS[category].some(p => lower.includes(p));
}

export function checkFormCompleteness(schema) {
  const type = detectType(schema.title);
  const required = type === 'default' ? ['contact', 'identity'] : (TYPE_REQUIREMENTS[type] ?? []);

  return required
    .filter(cat => !schema.questions.some(q => questionCovers(q, cat)))
    .map(cat => ({
      category: cat,
      message: `you might want a ${cat} field — want me to add one?`
    }));

// forms/google-forms-api.js
const BASE = 'https://forms.googleapis.com/v1';

function toGoogleItem(question, index) {
  const item = { title: question.title, questionItem: { question: { required: question.required ?? false } } };
  const q = item.questionItem.question;

  switch (question.type) {
    case 'SHORT_ANSWER': q.textQuestion = { paragraph: false }; break;
    case 'PARAGRAPH': q.textQuestion = { paragraph: true }; break;
    case 'MULTIPLE_CHOICE': q.choiceQuestion = { type: 'RADIO', options: question.options.map(v => ({ value: v })) }; break;
    case 'CHECKBOX': q.choiceQuestion = { type: 'CHECKBOX', options: question.options.map(v => ({ value: v })) }; break;
    case 'DROPDOWN': q.choiceQuestion = { type: 'DROP_DOWN', options: question.options.map(v => ({ value: v })) }; break;
    case 'LINEAR_SCALE':
      q.scaleQuestion = {
        low: question.scale?.low ?? 1, high: question.scale?.high ?? 5,
        lowLabel: question.scale?.lowLabel ?? '', highLabel: question.scale?.highLabel ?? ''
      };
      break;
    case 'DATE': q.dateQuestion = { includeTime: false }; break;
  }

  return item;
}

export async function getAuthToken() {
  return new Promise((resolve, reject) => {
    chrome.runtime.sendMessage({ type: 'GET_AUTH_TOKEN' }, (response) => {
      if (chrome.runtime.lastError || response?.error) {
        reject(new Error(response?.error ?? chrome.runtime.lastError?.message));
      } else {
        resolve(response.token);
      }
    });
  });
}

export class GoogleFormsApi {
  constructor(token) {
    this.token = token;
  }

  _headers() {
    return { 'Authorization': `Bearer ${this.token}`, 'Content-Type': 'application/json' };
  }

  async createForm(title, description = '') {
    const res = await fetch(`${BASE}/forms`, {
      method: 'POST', headers: this._headers(),
      body: JSON.stringify({ info: { title, documentTitle: title, description } })
    });
    if (!res.ok) throw new Error(`Create form failed: ${res.status}`);
    return res.json();
  }

  async addQuestion(formId, question, index) {
    return this.batchUpdate(formId, [{
      createItem: { item: toGoogleItem(question, index), location: { index } }
    }]);
  }

  async getForm(formId) {
    const res = await fetch(`${BASE}/forms/${formId}`, { headers: this._headers() });
    if (!res.ok) throw new Error(`Get form failed: ${res.status}`);
    return res.json();
  }

  async batchUpdate(formId, requests) {
    const res = await fetch(`${BASE}/forms/${formId}:batchUpdate`, {
      method: 'POST', headers: this._headers(),
      body: JSON.stringify({ requests })
    });
    if (!res.ok) throw new Error(`Batch update failed: ${res.status}`);
    return res.json();
  }
}

      templateMatch ? buildTemplateContext(templateMatch) : null
    );

    // Get the form ID from the URL (null when on /forms/create — will be set after creation)
    let formId = location.pathname.match(/\/forms\/d\/([^/]+)\//)?.[1] ?? null;

    const token = await getAuthToken();
    const api = new GoogleFormsApi(token);

    // If on the create page, we need to create the form first
    if (!formId) {
      const created = await api.createForm(userPrompt.slice(0, 60));
      formId = created.formId;
      // Redirect to the edit page — blanc will re-trigger via pendingPrompt? No:
      // We already have the prompt in memory as "current session". Don't redirect.
      // Just proceed to add questions to the new formId.
    }

    const collectedQuestions = [];
    let buffer = '';

    for await (const chunk of streamWithCascade(systemPrompt, userPrompt, (name) => {
      blanc.setState('error', { errorType: 'auth', name });
    })) {
      buffer += chunk;
      const parsed = tryParsePartialQuestions(buffer);
      for (let i = collectedQuestions.length; i < parsed.length; i++) {
        const q = parsed[i];
        collectedQuestions.push(q);
        blanc.showQuestionAdded(q.title);
        await api.addQuestion(formId, q, i);
      }
    }

    const warnings = checkFormCompleteness({ title: userPrompt, questions: collectedQuestions });

    await saveFormToHistory({
      title: userPrompt.slice(0, 60),
      questionCount: collectedQuestions.length,
      templateUsed: templateMatch?.name ?? null,
      createdAt: new Date().toISOString()
    });
    await saveFieldSchema(userPrompt.slice(0, 20), collectedQuestions.map(q => q.title));

    blanc.setState('done', { warning: warnings[0]?.message ?? null });

    // Offer Slack share
    if (keys.slackWebhook) {
      const formUrl = `https://docs.google.com/forms/d/${formId}`;
      await postToSlack(keys.slackWebhook, {
        title: userPrompt.slice(0, 60),
        questionCount: collectedQuestions.length,
        editUrl: `${formUrl}/edit`,
        viewUrl: `${formUrl}/viewform`
      }).catch(() => {}); // non-fatal
    }

  } catch (err) {
    if (err.type === 'ALL_FAILED') {
      blanc.setState('error', { errorType: 'all_failed' });
    } else {
      blanc.setState('error', { errorType: 'parse' });
      console.error('[Formie]', err);
    }
  }
});
```

// service-worker.js
const STORAGE_DEFAULTS = {
  keys: { gemini: '', grok: '', slackWebhook: '' },
  memory: { fieldSchemas: [], history: [], savedTemplates: [] },
  pendingPrompt: null
};

chrome.runtime.onInstalled.addListener(() => {
  chrome.storage.local.get(STORAGE_DEFAULTS, (existing) => {
    chrome.storage.local.set(existing);
  });
});

chrome.runtime.onMessage.addListener((message, _sender, sendResponse) => {
  if (message.type === 'GET_AUTH_TOKEN') {
    chrome.identity.getAuthToken({ interactive: true }, (token) => {
      if (chrome.runtime.lastError) {
        sendResponse({ error: chrome.runtime.lastError.message });
      } else {
        sendResponse({ token });
      }
    });
    return true; // keep message channel open for async response
  }

  if (message.type === 'OPEN_TAB') {
    chrome.tabs.create({ url: message.url });
    sendResponse({ ok: true });
  }

  return true;
});
